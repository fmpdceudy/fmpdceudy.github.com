#ffmpeg内部锁管理（qt下）
    int cb(void **mutex, enum AVLockOp op)
    {
        switch(op){
        case AV_LOCK_CREATE:
            *mutex= ((void *)(new QMutex()));
            break;
        case AV_LOCK_OBTAIN:
            ((QMutex *)(*mutex))->lock();
            break;
       case AV_LOCK_RELEASE:
            ((QMutex *)(*mutex))->unlock();
            break;
        case AV_LOCK_DESTROY:
            delete (QMutex *)(*mutex);
            break;
        }
        return 0;
    }
    /*init*/
    av_lockmgr_register(&cb);
    avcodec_init();
    avcodec_register_all();
